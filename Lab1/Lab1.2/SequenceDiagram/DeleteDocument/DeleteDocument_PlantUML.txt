@startuml DeleteDocument
actor Admin as User
participant System
participant Server

activate User
User -> System: Access "Document Information" screen
activate System
User -> System: Click on "Delete Document" button 
System -> User: Ask again(message)
alt Agree
    System --> Server: Send "Delete" request
    activate Server
    Server -> Server: Delete Document and Save changes
    Server --> System: Delete successfully
    deactivate Server
    System --> User: Move to "Document List" screen
else Deny
    System --> User: Move user to "Document Information" screen
    deactivate System
end
deactivate User

@enduml
