@startuml DeleteStaff
actor Admin as User
participant System
participant Server

activate User
User -> System: Access "Staff Information" screen
activate System
User -> System: Click on "Delete Staff" button 
System -> User: Ask again(message)
alt Agree
    System --> Server: Send "Delete" request
    activate Server
    Server -> Server: Delete Staff and Save changes
    Server --> System: Delete successfully
    deactivate Server
    System --> User: Move to "Staff List" screen
else Deny
    System --> User: Move user to "Staff Information" screen
    deactivate System
end
deactivate User

@enduml
