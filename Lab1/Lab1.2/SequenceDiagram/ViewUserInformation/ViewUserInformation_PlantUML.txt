 @startuml ViewUserInformation
actor Admin_User as User
participant System
participant Server

activate User
User -> System: Access "User Information" screen
activate System
System -> Server: Send "View user information" request
activate Server
Server -> Server: Prepare user information
alt Valid user
    Server --> System: Request Successful
    Server --> System: User information
    System --> User: Show user information on screen
else Invalid user
    Server --> System: Request failed
    deactivate Server
    System --> User: Show error notification
    deactivate System
end 
deactivate User

@enduml
