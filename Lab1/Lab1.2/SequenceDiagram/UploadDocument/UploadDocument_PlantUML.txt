@startuml UploadDocument
actor Admin_Staff_User as User
participant System
participant Server

activate User
User -> System: Access "Document List" screen
activate System
User -> System: Click on "Upload Document" button 
System -> User: Ask Document Location
User -> System: Choose and Rename Document
User -> System: Click "Upload" button
System -> Server: Send request 
activate Server
Server -> Server: Save Document in temporary location
Server -> Server: Authenticate Document 
alt Accept
    Server -> Server: Save Document
    Server --> System: Request Accepted
    System --> User: Upload succcessfully(message)
    System --> User: Move user to "Document List" screen
else Deny
    Server -> Server: Delete Document
    Server --> System: Request Denied
    deactivate Server
    System --> User: Upload failed(message)
    System --> User: Move user to "Document List" screen
    deactivate System
end 
deactivate User

@enduml
